Select  buz.name,
        buz.stars,
        time(substr(h.hours, instr(h.hours,'-')+1)) - time(substr(h.hours, instr(h.hours,'|')+1, instr(h.hours,'-')-instr(h.hours,'|')-1)) as opening_hours
from business as buz
join hours as h
on h.business_id=buz.id
where opening_hours is not null)
group by name
